27  value += rhs.value;
28  return *this;

32  value -= rhs.value;
33  return *this;
34  }
35

37

39  value *= rhs;
40  return *this;
41  }
42 
43  constexpr RQuantity const &operator/=(const double rhs) {
44  value /= rhs;
45  return *this;
46  }
47 
48  // Returns the value of the quantity in multiples of the specified unit
49  constexpr double convert(const RQuantity &rhs) const { return value / rhs.value; }
50 
51  // returns the raw value of the quantity (should not be used)
52  constexpr double getValue() const { return value; }
53

55  return RQuantity<MassDim, LengthDim, TimeDim, AngleDim>(std::fabs(value));

59  std::ratio_divide<TimeDim, std::ratio<2>>, std::ratio_divide<AngleDim, std::ratio<2>>>
60  sqrt() const {

63  std::sqrt(value));
